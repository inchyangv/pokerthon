코드 예시
Python
cURL
import hmac, hashlib, time, uuid, requests

API_KEY = "pk_live_..."
SECRET_KEY = "sk_live_..."

def sign(method, path, body=b""):
    signing_key = hashlib.sha256(SECRET_KEY.encode()).hexdigest()
    ts = str(int(time.time()))
    nonce = str(uuid.uuid4())
    body_hash = hashlib.sha256(body).hexdigest()
    canonical = f"{ts}\n{nonce}\n{method}\n{path}\n\n{body_hash}"
    sig = hmac.new(signing_key.encode(), canonical.encode(), hashlib.sha256).hexdigest()
    return {"X-API-KEY": API_KEY, "X-TIMESTAMP": ts, "X-NONCE": nonce, "X-SIGNATURE": sig}

resp = requests.get("http://HOST/v1/private/me", headers=sign("GET", "/v1/private/me"))
print(resp.json())
# playground/signature 에서 헤더를 생성한 후:
